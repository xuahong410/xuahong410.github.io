Vue3里面使用univer

1. npm install @univerjs/presets @univerjs/core
2. Npm install rxjs
3. 在线表格组件

<template>

  <!-- 1. 挂载容器（必须指定高度） -->

  <div ref="container" class="univer-sheet-container"></div>

</template>

 

<script setup>

import { ref, onMounted, onBeforeUnmount } from 'vue';

import { createUniver, LocaleType, mergeLocales, UniverInstanceType } from '@univerjs/presets';

// 导入核心表格预设（包含基础编辑、公式、样式）

import { UniverSheetsCorePreset } from '@univerjs/preset-sheets-core';

// 导入中文语言包

import UniverPresetSheetsCoreZhCN from '@univerjs/preset-sheets-core/locales/zh-CN';

// 导入基础样式（必须）

import '@univerjs/preset-sheets-core/lib/index.css';

 

const container = ref(null);

let univerInstance = null;

let univerAPI = null;

 

onMounted(() => {

  if (!container.value) return;

 

  // 2. 初始化 Univer 实例（中文环境）

  const { univer, univerAPI: api } = createUniver({

    locale: LocaleType.ZH_CN,

    locales: {

      [LocaleType.ZH_CN]: mergeLocales(

        UniverPresetSheetsCoreZhCN,

      ),

    },

    // 注册核心表格预设

    presets: [

      UniverSheetsCorePreset({

        container: container.value, // 绑定 DOM 元素

      }),

    ],

  });

 

  univerInstance = univer;

  univerAPI = api;

 

  // 3. 创建空白工作簿（或加载后端数据）

  const workbookData = {

    id: 'tax-basic-workbook',

    name: '税务基础表',

    sheetOrder: ['sheet1'],

    sheets: {

      sheet1: {

        id: 'sheet1',

        name: 'Sheet1',

        cellData: {

          // 表头

          0: {

            0: { v: '项目', s: { bl: 1 } }, // s 为样式（加粗）

            1: { v: '金额' },

            2: { v: '税率' }

          },

          // 数据行

          1: { 0: { v: '增值税' }, 1: { v: 10000 }, 2: { v: 0.13 } },

          2: { 0: { v: '所得税' }, 1: { v: 5000 }, 2: { v: 0.25 } },

          // 公式行（SUM 计算）

          3: { 0: { v: '合计' }, 1: { f: '=SUM(B2:B3)' } },

        },

      },

    },

  };

 

  // 创建表格单元

  univer.createUnit(UniverInstanceType.UNIVER_SHEET, workbookData);

});

 

onBeforeUnmount(() => {

  // 4. 销毁实例（防止内存泄漏）

  if (univerInstance) {

    univerInstance.dispose();

  }

});

</script>

 

<style scoped>

.univer-sheet-container {

  width: 100%;

  height: 600px; /* 必须设置高度，否则表格无法渲染 */

  border: 1px solid #e8e8e8;

}

</style>
